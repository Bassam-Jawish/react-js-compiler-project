    } else {
      const results = items.filter((item) =>
        item.name.toLowerCase().includes(searchTerm.toLowerCase())
      );
      setItems(results);
    }
  }, [searchTerm]);
  
  return (
    <div className="search-bar">
      <input
        id = "searchInput"
        type="text"
        placeholder="Search..."
        value={searchTerm}
        onChange={() => setSearchTerm(event.target.value)}
      />
    </div>
  );
}
